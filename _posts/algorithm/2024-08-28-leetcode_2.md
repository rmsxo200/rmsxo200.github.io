# 릿코드 문제풀이


```
//String s = "LVIII"; //58
//String s = "III";
String s = "MCMXCIV"; // 1994
int sum = 0;
Map<Character, Integer> romanMap = new HashMap<Character, Integer>(){
    {
        put('I', 1);
        put('V', 5);
        put('X', 10);
        put('L', 50);
        put('C', 100);
        put('D', 500);
        put('M', 1000);
    }
};

for (int i = 0; i < s.length(); i++) {
    int thisNum = romanMap.get(s.charAt(i));
    int nextNum = romanMap.get(s.charAt(i+1));
    
    if(i != s.length()-1) {//마지막이 아니면
        if(thisNum < nextNum) {
            sum += nextNum - thisNum;
            i++; //1증가
        } else {
            sum += thisNum;
        }
    } else { //마지막
        sum += thisNum;
    }
}
```
